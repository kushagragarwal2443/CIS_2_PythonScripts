import numpy as np
import scipy as sp
import matplotlib.pyplot as plt

def areacalculate(a2,a1,a0,a,b,N):
    hmax=(b-a)/N
    #h max is used to calculate h for the respective integral
    calc=0
    i=a
    while True:
        #infinite loop
        h=hmax/(1+abs(functionddx(a2,a1,i)))
        #this makes h small for large values of differential
        #this makes h variable and therefore accomodates for adaptive size grids
        if i==a :
            calc+=0.5*h*function(a2,a1,a0,i)
            #boundary values have a coefficient of 0.5 multiplied
        else:
            calc+=h*function(a2,a1,a0,i)
        i+=h
        if i>=b:
            break
    return calc
def function(a2,a1,a0,x):
    #calculating the value at certain x
    return a2*x*x+a1*x+a0
def functionddx(a2,a1,x):
    #this function returns the differential of the function at x
    return 2*x*a2 + a1
