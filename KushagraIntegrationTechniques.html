Grid schemes¶

Typically used for 1/2 dimensional integrals, by evaluating the value of the function on a regular grid and then using approximation schemes like 'trapezoidal rule'

import numpy as np
import scipy as sp
import matplotlib.pyplot as plt

def calculation(f,a,b,N):
    h=(b-a)/N
    #h is the equivalent of dx in integration
    calc=0
    for i in np.arange(a,b,h):
        calc+=h*f(i)
    calc+=0.5*h*f(a)
    #i started from a+h and not a so we added 0.5 times the integral at tha point
    calc-=0.5*h*f(b)
    #b was included in the calculation but we need only 0.5 times of the boundary values so we deduct
    return calc
def parabola(x):
    #this function returns 2*value of y as it is returns two values for y for each x
    return 2*np.sqrt(4*x)
def circle(x):
    #similarly for the circle we multiply by two as it is symmetric about y=0 here
    return 2*np.sqrt(1-x*x)

import numpy as np
import scipy as sp
import matplotlib.pyplot as plt

def calculate(f,f1,a,b,N):
    hmax=(b-a)/N
    #h max is used to calculate h for the respective integral
    calc=0
    i=a
    while True:
        #infinite loop
        h=hmax/(1+abs(f1(i)))
        #this makes h small for large values of differential
        #this makes h variable and therefore accomodates for adaptive size grids
        if i==a :
            calc+=0.5*h*f(i)
            #boundary values have a coefficient of 0.5 multiplied
        else:
            calc+=h*f(i)
        i+=h
        if i>=b:
            break
    return calc
def parabola(x):
    return 2*np.sqrt(4*x)
def circle(x):
    return 2*np.sqrt(1-x*x)
def paraboladdx(x):
    #this function returns the differential of the function at x
    z=parabola(x)
    if z!=0:
        return 2/(z)
    else :
        #if the differential turns out to be infinity we return a large number instead 
        return 1000000    
def circleddx(x):
    z=circle(x)
    if z!=0:
        return -1*x*(1/circle(x))
    else:
        return 1000000

import numpy as np
import matplotlib.pyplot as plt

def MonteCarlo(a,b,N):
    count=0
    for i in range(0,N):
        x1=np.random.uniform(0,1,1)
        #stores a random value in x1
        x2=np.random.uniform(0,1,1)
        #stores a random value in x2
        z=(x1*x1)+(x2*x2)
        #if it satisfies equation of disc of unit radius
        if z<=1 :
            count=count+1
    return (count/N)
#the function should return ratio of area of circle to the area of rectangle
# an explanation for it is as follows- we throw pebbles of negligible size
#onto a square of unit radius and mark a quarter circle on it
#the ratio of pebbles inside the quarter to the total area gives the effective area of the quarter = pi/4

def MC(a,b,N):
    
    count=0
    #arrays created of size N
    x1=[0]*N
    x2=[0]*N
    x3=[0]*N
    x4=[0]*N
    s=0
    for i in range(1,N):
        points=np.random.uniform(a,b,2)
        x1[i-1]=points[0]
        x2[i-1]=points[1]
        z=(x1[i-1]*x1[i-1])+(x2[i-1]*x2[i-1])
        if z<=a*a :
            count=count+1
            #if the point falls in the circle

            x3[s]=x1[i-1]
            x4[s]=x2[i-1]

    return count/N
